var props = new java.util.Properties();
var is = new java.io.FileInputStream(file);
props.load(is);
is.close();
for(var it = props.entrySet().iterator(); it.hasNext();)
{
    var entry = it.next();
    vars.put(entry.getKey(), entry.getValue());
}
